P1.1 Исправить getStats() — убрать SELECT * + JS-фильтрацию models.service.ts:279 · profiles.service.ts:485 Critical ⏱ 30 мин
Сейчас оба сервиса грузят ВСЕ строки таблицы model_profiles в память Node.js и считают длины массивов. При 1000+ моделях — катастрофа. Заменяем на один SQL-запрос с COUNT + FILTER.
1. apps/api/src/models/models.service.ts — строка 273
Добавить импорт count, sql из drizzle-orm вверху файла, затем заменить метод.
// Добавить в импорты (вверху файла)
import { count, sql } from 'drizzle-orm';

// Заменить метод getStats() полностью
async getStats() {
  const all = await this.db.select().from(modelProfiles);
  return {
    total: all.length,
    online: all.filter((m) => m.availabilityStatus === 'online').length,
    verified: all.filter((m) => m.verificationStatus === 'verified').length,
    elite: all.filter((m) => m.eliteStatus === true).length,
  };
}

async getStats() {
  const [row] = await this.db
    .select({
      total:    count(),
      online:   sql<number>`count(*) filter (where ${modelProfiles.availabilityStatus} = 'online')`,
      verified: sql<number>`count(*) filter (where ${modelProfiles.verificationStatus} = 'verified')`,
      elite:    sql<number>`count(*) filter (where ${modelProfiles.eliteStatus} = true)`,
    })
    .from(modelProfiles);
  return {
    total:    Number(row.total),
    online:   Number(row.online),
    verified: Number(row.verified),
    elite:    Number(row.elite),
  };
}

P1.2 Параллельные fetch через Promise.all() на /models и /models/[slug] app/models/page.tsx · app/models/[slug]/page.tsx High ⏱ 45 мин
Два последовательных fetch = 2× latency. Запускаем оба запроса одновременно.
1. app/models/page.tsx — загрузка моделей + статистики
// Было — последовательно:
const models = await fetchModels();
const stats  = await fetchStats();

// Стало — параллельно:
const [models, stats] = await Promise.all([
  fetchModels(),
  fetchStats(),
]);
2. app/models/[slug]/page.tsx — профиль + медиафайлы
const profile = await fetchProfile(slug);
const media   = await fetchMedia(profile.id);

const profile = await fetchProfile(slug);
// media зависит от profile.id — первый запрос оставляем, второй запускаем параллельно
const [fullProfile, media] = await Promise.all([
  fetchProfileDetails(profile.id),
  fetchMedia(profile.id),
]);
Если второй запрос зависит от результата первого (нужен id) — только первый последовательный, остальные в Promise.all. Не форсировать параллелизм там где есть реальная зависимость.
Проверка Network tab в DevTools: оба запроса стартуют одновременно (одинаковый timestamp), а не друг за другом.

P2.3 Исправить token refresh — убрать блокирующий UI freeze apps/web/lib/api-client.ts:234 High ⏱ 2 ч
При 401 сейчас запускается синхронный refresh — все параллельные запросы блокируются и ждут. Нужен паттерн "refresh once, queue the rest".
// apps/web/lib/api-client.ts

let isRefreshing = false;
let refreshQueue: Array<(token: string) => void> = [];

async function refreshTokenOnce(): Promise<string> {
  if (isRefreshing) {
    // встаём в очередь, ждём пока первый refresh завершится
    return new Promise((resolve) => {
      refreshQueue.push(resolve);
    });
  }
  isRefreshing = true;
  try {
    const newToken = await doRefreshRequest();
    refreshQueue.forEach((cb) => cb(newToken));
    return newToken;
  } finally {
    isRefreshing = false;
    refreshQueue = [];
  }
}

// В interceptor на 401:
if (response.status === 401) {
  const token = await refreshTokenOnce(); // ждёт один раз, очередь обслуживается
  return fetch(request, { headers: { Authorization: `Bearer ${token}` } });
}
Проверка Открыть DevTools → Network, вручную истечь access token. При следующем запросе: ровно один запрос к /auth/refresh, остальные запросы стартуют сразу после него (не ждут по очереди).

P3.1 Конвертировать /models в Server Component app/models/page.tsx Critical ⏱ 4–6 ч
Сейчас: браузер получает пустую страницу, запускает JS, делает fetch, получает данные, рендерит. При SSR: сервер уже отдаёт готовый HTML. Пользователь видит контент мгновенно.
Стратегия разделения
Не весь page.tsx может быть серверным. Разбиваем по принципу: данные — сервер, интерактивность — клиент.
// БЫЛО: один большой 'use client' компонент
'use client';
export default function ModelsPage() {
  const [models, setModels] = useState([]);
  useEffect(() => { fetchModels().then(setModels); }, []);
  return <ModelList models={models} />;
}

// СТАЛО: серверный page + клиентские острова

// app/models/page.tsx — SERVER component (нет 'use client')
import { ModelGrid }    from '@/components/models/ModelGrid';       // server
import { FilterPanel }  from '@/components/models/FilterPanel';      // client
import { fetchPublicModels, fetchModelStats } from '@/lib/api-server';

export default async function ModelsPage({ searchParams }) {
  const [models, stats] = await Promise.all([
    fetchPublicModels(searchParams),
    fetchModelStats(),
  ]);

  return (
    <main>
      <FilterPanel />           {/* 'use client' — фильтры и стейт */}
      <ModelGrid models={models} stats={stats} />  {/* server — статичный список */}
    </main>
  );
}
Что оставить клиентским ('use client')
  • Фильтры / поиск (useState, URL params)
  • Избранное (ModelFavoriteButton — уже есть)
  • Лайтбокс / галерея
  • GuestBookingModal / BookingTonModal
Серверный API-клиент
// apps/web/lib/api-server.ts — новый файл (только для server components)
const API = process.env.INTERNAL_API_URL ?? 'http://localhost:3000';

export async function fetchPublicModels(params = {}) {
  const qs = new URLSearchParams(params as Record<string,string>).toString();
  const res = await fetch(`${API}/models/public?${qs}`, {
    next: { revalidate: 60 },   // ISR: кеш 60 секунд
  });
  if (!res.ok) throw new Error('Failed to fetch models');
  return res.json();
}
INTERNAL_API_URL на VPS = http://127.0.0.1:3000 (не через Nginx). Сервер обращается к API напрямую — быстрее и без TLS overhead.
Проверка curl -s http://localhost:3001/models | grep "<article" — карточки моделей должны быть в HTML ответе (не пусто). Lighthouse → FCP < 1.5s.
