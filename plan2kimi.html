❓ 4. Architecture Questions (Требуют решения)

Q1: Authentication Strategy — Token Blacklist

Проблема: Требуется ли server-side token blacklist для logout (хранение revoked tokens в Redis) или достаточно client-side token removal?

Option A: Client-side only (проще, но токены действительны до экспирации)
Option B: Redis blacklist (безопаснее, но требует Redis integration)
Recommendation: Option B для premium платформы

Q2: Role Hierarchy & Permissions

Проблема: Может ли Manager редактировать профили Model, или только approve/reject? Нужна ли granular permission system?

Option A: Simple 4 roles (Admin, Manager, Model, Client)
Option B: Granular permissions (canEdit, canDelete, canViewPhone, canModerateMedia)
Recommendation: Option A для MVP, Option B для scale

Q3: Data Retention (152-ФЗ Compliance)

Проблема: По 152-ФЗ требуется хранить audit logs 5 лет. Храним в PostgreSQL или выгружаем в ELK/ClickHouse?

Option A: PostgreSQL audit_logs table (проще, но раздувает БД)
Option B: ELK Stack (сложнее, но лучше для analytics)
Recommendation: Option A для MVP с партиционированием по месяцам

Q4: Anti-Leak Strictness

Проблема: При обнаружении попытки слива контактов (3 strikes):

Option A: Masking (замена цифр на *)
Option B: Block message + warn admin
Option C: Auto-blacklist client
Recommendation: Option B → C после 3 нарушений

Q5: Payment Resilience

Проблема: Нужна ли circuit breaker logic для платежных шлюзов?

Option A: Single provider (YooKassa) + retry
Option B: Fallback chain (YooKassa → Robokassa → Cryptomus)
Recommendation: Option A для MVP, Option B для production

Q6: Rate Limiting Thresholds

Проблема: Какие лимиты установить для разных endpoint'ов?

Auth endpoints: 5 requests/min (brute-force protection)
General API: 100 requests/min (normal usage)
Strict operations: 10 requests/min (expensive ops)
Public endpoints: 200 requests/min (catalog, health)

JWT Guard Implementation

// apps/api/src/auth/guards/jwt-auth.guard.ts
import {
  CanActivate,
  ExecutionContext,
  Injectable,
  UnauthorizedException,
} from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { Request } from 'express';
import { ConfigService } from '@nestjs/config';

export interface JwtPayload {
  sub: string;          // User ID
  email: string;
  role: 'admin' | 'manager' | 'model' | 'client';
  jti: string;          // JWT ID (session)
  iat: number;
  exp: number;
}

@Injectable()
export class JwtAuthGuard implements CanActivate {
  constructor(
    private jwtService: JwtService,
    private configService: ConfigService,
  ) {}

  async canActivate(context: ExecutionContext): Promise {
    const request = context.switchToHttp().getRequest();
    const token = this.extractTokenFromHeader(request);

    if (!token) {
      throw new UnauthorizedException('Authentication token missing');
    }

    try {
      const payload = await this.jwtService.verifyAsync(token, {
        secret: this.configService.getOrThrow('JWT_SECRET'),
        algorithms: ['HS256'],
        issuer: 'lovnge-api',
        audience: 'lovnge-client',
      });

      // TODO: Check Redis blacklist here
      // const isBlacklisted = await this.redis.get(`blacklist:${payload.jti}`);
      // if (isBlacklisted) throw new UnauthorizedException('Token revoked');

      request['user'] = {
        userId: payload.sub,
        email: payload.email,
        role: payload.role,
        sessionId: payload.jti,
        iat: payload.iat,
        exp: payload.exp,
      };

      return true;
    } catch (error) {
      if (error.name === 'TokenExpiredError') {
        throw new UnauthorizedException('Token expired. Please refresh.');
      }
      if (error.name === 'JsonWebTokenError') {
        throw new UnauthorizedException('Invalid token format');
      }
      throw new UnauthorizedException('Authentication failed');
    }
  }

  private extractTokenFromHeader(request: Request): string | undefined {
    const [type, token] = request.headers.authorization?.split(' ') ?? [];
    return type === 'Bearer' ? token : undefined;
  }
}

Roles Guard Implementation

// apps/api/src/auth/guards/roles.guard.ts
import {
  CanActivate,
  ExecutionContext,
  Injectable,
  ForbiddenException,
} from '@nestjs/common';
import { Reflector } from '@nestjs/core';

export const ROLES_KEY = 'roles';

@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  async canActivate(context: ExecutionContext): Promise {
    const requiredRoles = this.reflector.getAllAndOverride(ROLES_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);

    if (!requiredRoles) {
      return true; // No role restriction
    }

    const request = context.switchToHttp().getRequest();
    const user = request['user'];

    if (!user) {
      throw new ForbiddenException('User not authenticated');
    }

    // Role hierarchy: admin > manager > model > client
    const roleHierarchy = {
      admin: 4,
      manager: 3,
      model: 2,
      client: 1,
    };

    const userLevel = roleHierarchy[user.role] || 0;
    const requiredLevel = Math.max(
      ...requiredRoles.map(role => roleHierarchy[role] || 0)
    );

    if (userLevel < requiredLevel) {
      throw new ForbiddenException('Insufficient permissions');
    }

    return true;
  }
}

// Usage in controller:
// @Roles('admin', 'manager')
// @Put('media/:id/approve')
// async approveMedia(@Param('id') mediaId: string) { ... }

Rate Limiting Configuration

// apps/api/src/security/rate-limit.config.ts
import { Module, Global } from '@nestjs/common';
import { ThrottlerModule } from '@nestjs/throttler';
import { APP_GUARD } from '@nestjs/core';
import { ThrottlerGuard } from '@nestjs/throttler';

@Global()
@Module({
  imports: [
    ThrottlerModule.forRoot([
      {
        name: 'default',
        ttl: 60000, // 1 minute
        limit: 100, // 100 requests per minute
      },
      {
        name: 'auth',
        ttl: 60000,
        limit: 5, // 5 login attempts per minute
      },
      {
        name: 'strict',
        ttl: 60000,
        limit: 10, // 10 requests for expensive ops
      },
    ]),
  ],
  providers: [
    {
      provide: APP_GUARD,
      useClass: ThrottlerGuard,
    },
  ],
  exports: [ThrottlerModule],
})
export class RateLimitModule {}

// apps/api/src/profiles/profiles.service.ts
async getCatalog(filters?: CatalogFiltersDto): Promise {
  const conditions: any[] = [eq(modelProfiles.isPublished, true)];

  // WHITELIST sortable columns
  const sortableColumns = [
    'ratingReliability',
    'createdAt',
    'displayName',
    'age',
    'height',
  ] as const;
  type SortableColumn = typeof sortableColumns[number];

  const orderByColumn: SortableColumn =
    filters?.orderBy && sortableColumns.includes(filters.orderBy)
      ? filters.orderBy
      : 'displayName';

  const orderDirection = filters?.order === 'asc' ? 'asc' : 'desc';

  // VALIDATE numeric filters
  const limit = Math.min(Math.max(filters?.limit || 50, 1), 100);
  const offset = Math.max(filters?.offset || 0, 0);

  // VALIDATE numeric ranges
  if (filters?.minAge) {
    conditions.push(gte(modelProfiles.age, Math.min(Math.max(filters.minAge, 18), 45)));
  }
  if (filters?.maxAge) {
    conditions.push(lte(modelProfiles.age, Math.min(Math.max(filters.maxAge, 18), 45)));
  }

  return this.db
    .select()
    .from(modelProfiles)
    .where(and(...conditions))
    .orderBy(
      orderDirection === 'asc'
        ? asc(modelProfiles[orderByColumn])
        : desc(modelProfiles[orderByColumn])
    )
    .limit(limit)
    .offset(offset);
}
